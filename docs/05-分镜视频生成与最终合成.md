## 二、Phase A — 分镜图生成

### 2.1 提示词构建

对每个分镜 shot，合成完整的图像生成提示词：

```typescript
function buildShotImagePrompt(
  shot: ScriptAnalysis["episodes"][0]["scenes"][0]["shots"][0],
  sceneInfo: ScriptAnalysis["scenes"][0],
  characterInfo: ScriptAnalysis["characters"],
  styleConfig: StyleConfig,
): string {
  // 1. shot 自身的视觉描述
  const shotDesc = shot.visual_prompt || shot.action;

  // 2. 景别映射
  const shotSizeMap: Record<string, string> = {
    "ELS": "extreme long shot, vast landscape",
    "LS": "long shot, full environment visible",
    "FS": "full shot, character full body in frame",
    "MS": "medium shot, character from waist up",
    "CU": "close-up shot, face and shoulders",
    "ECU": "extreme close-up, eyes or detail fill frame",
  };
  const shotSize = shotSizeMap[shot.shot_type] || "medium shot";

  // 3. 场景环境
  const env = `${sceneInfo.environment.time_of_day}, ${sceneInfo.environment.weather}, ${sceneInfo.environment.atmosphere}`;

  // 4. 角色外貌关键词（从锁定的资产中提取）
  const charKeywords = shot.subject; // 已在剧本解析中填入角色名+动作

  // 5. 风格关键词
  const styleKw = extractTopKeywords(styleConfig.compiled_style_prompt, 6);

  // 合成
  return [
    shotDesc,
    shotSize,
    charKeywords,
    env,
    sceneInfo.lighting.mood,
    styleKw,
    "cinematic composition, high quality",
  ].filter(Boolean).join(", ");
}
```

```typescript
async function generateStoryboardImage(
  shot: ShotWithMeta,
  projectId: string,
  imageModel: string = "nano-banana-2",
): Promise<string | null> {
  const prompt = buildShotImagePrompt(shot, shot.scene, shot.characters, shot.styleConfig);

  const sizeMap: Record<string, string> = {
    "16:9": "1536x1024",
    "9:16": "1024x1536",
    "1:1": "1024x1024",

    prompt,
    n: 1,
    size,
    project_id: projectId,
    asset_filename: `storyboard_ep${shot.episode_id}_s${shot.shot_index}.png`,
  });

  return result.saved_assets?.[0]?.asset_url || result.data?.[0]?.url || null;
}
```

### 3.1 视频生成策略

使用现有的 **Seedance 2.0** 视频生成 API（通过 `/api/tasks`）。

对每个分镜，以分镜图作为起始帧，生成视频片段：

```typescript
async function generateShotVideo(
  shot: WfShot,
  storyboardImage: string,         // 分镜图 URL
  project: WfProject,
): Promise<{ taskId: string }> {

  // 构建视频生成提示词
  const videoPrompt = buildVideoPrompt(shot, project);

  // 选择生成模式
  const mode = storyboardImage ? "first_frame" : "text";

  const params: GenerationParams = {
    mode,
    model: "doubao-seedance-2.0",
    resolution: "720p",
    ratio: project.style_config!.aspect_ratio,
    duration: shot.duration || 5,
    cameraPreset: mapCameraMovement(shot.camera_movement),
    motionSpeed: "steady",
    generateAudio: false,           // 后期单独处理音频
    firstFrame: storyboardImage,
  };

  const payload = buildPayload(videoPrompt, params);
  const result = await createTask(payload, {
    title: `EP${shot.episode_id}_Shot${shot.id}`,
    mode: "storyboard_shot",
  });

  return { taskId: result.id };
}
```

```typescript
function buildVideoPrompt(shot: WfShot, project: WfProject): string {
  const styleKw = extractTopKeywords(project.style_config!.compiled_style_prompt, 5);

  const parts = [
    shot.raw_description || shot.prompt,        // 画面描述
    `camera: ${shot.camera_movement}`,          // 运镜
    `character action: ${shot.prompt}`,         // 角色动作
  ];

  // 添加运镜辅助词
  const camera = CAMERA_PRESETS.find(c => c.value === mapCameraMovement(shot.camera_movement));
  if (camera?.helper) parts.push(camera.helper);

  // 添加风格
  parts.push(styleKw);

  return parts.filter(Boolean).join("\n");
}

function mapCameraMovement(movement: string): string {
  const map: Record<string, string> = {
    "push_in_slow": "push_in",
    "pull_back": "pull_back",
    "pan_left": "pan",
    "pan_right": "pan",
    "orbit": "orbit",
    "static": "auto",
    "handheld": "handheld",
    "tilt_up": "push_in",
    "crane_down": "pull_back",
  };
  return map[movement] || "auto";
}
```

### 3.3 视频生成队列

```typescript
async function generateAllVideos(project: WfProject): Promise<void> {
  const allShots = flattenShots(project);
  const videoTasks: { shot: WfShot; taskId: string }[] = [];

  // Phase 1: 提交所有生成任务
  for (const shot of allShots) {
    if (shot.status === "done") continue;  // 跳过已完成的

    const { taskId } = await generateShotVideo(
      shot,
      shot.storyboard_image!,
      project,
    );

    videoTasks.push({ shot, taskId });
    shot.video_task_id = taskId;
    shot.status = "filming";

    // 每提交一个任务等 2 秒，避免 rate limit
    await sleep(2000);
  }

  // Phase 2: 轮询所有任务直到完成
  await pollVideoTasks(videoTasks, project);
}

async function pollVideoTasks(
  tasks: { shot: WfShot; taskId: string }[],
  project: WfProject,
): Promise<void> {
  const pending = new Set(tasks.map(t => t.taskId));

  while (pending.size > 0) {
    for (const task of tasks) {
      if (!pending.has(task.taskId)) continue;

      try {
        const result = await queryTask(task.taskId);
        const progress = deriveProgress(result);

        if (progress.stage === 3) {  // 完成
          pending.delete(task.taskId);

          if (result.status === "succeeded") {
            task.shot.video_url = extractVideoUrl(result);
            task.shot.status = "done";
          } else {
            task.shot.status = "failed";
          }

          // 保存进度
          await wfUpdateProject(project.id, { episodes: project.episodes });
        }
      } catch {}
    }

    if (pending.size > 0) {
      await sleep(5000);  // 每 5 秒轮询一次
    }
  }
}
```

### 4.2 字幕生成

从剧本解析的对白数据生成 SRT 字幕：

```python
# video_composer.py — 字幕生成

def generate_srt(episodes: list, shots_with_timing: list) -> str:
    """从分镜对白生成 SRT 字幕文件"""
    srt_entries = []
    index = 1
    current_time = 0.0

    for shot in shots_with_timing:
        if not shot.get("dialogue"):
            current_time += shot["duration"]
            continue

        start = format_srt_time(current_time)
        end = format_srt_time(current_time + shot["duration"] * 0.8)  # 字幕提前消失

        srt_entries.append(
            f"{index}\n"
            f"{start} --> {end}\n"
            f"{shot['dialogue']}\n"
        )
        index += 1
        current_time += shot["duration"]

    return "\n".join(srt_entries)

def format_srt_time(seconds: float) -> str:
    h = int(seconds // 3600)
    m = int((seconds % 3600) // 60)
    s = int(seconds % 60)
    ms = int((seconds % 1) * 1000)
    return f"{h:02d}:{m:02d}:{s:02d},{ms:03d}"
```

### 4.3 视频拼接（FFmpeg）

```python
# video_composer.py — FFmpeg 拼接

import subprocess
import tempfile

def compose_video(
    shot_videos: list[str],    # 本地视频文件路径列表
    output_path: str,
    subtitle_srt: str | None = None,
    bgm_path: str | None = None,
) -> str:
    """将分镜视频拼接为完整影片"""

    # 1. 创建 concat 列表文件
    with tempfile.NamedTemporaryFile(mode='w', suffix='.txt', delete=False) as f:
        for video in shot_videos:
            f.write(f"file '{video}'\n")
        concat_file = f.name

    # 2. 基础拼接
    cmd = [
        "ffmpeg", "-y",
        "-f", "concat", "-safe", "0",
        "-i", concat_file,
        "-c:v", "libx264",
        "-preset", "medium",
        "-crf", "18",
        "-r", "24",
        "-pix_fmt", "yuv420p",
    ]

    # 3. 烧录字幕（可选）
    if subtitle_srt:
        srt_path = output_path.replace(".mp4", ".srt")
        Path(srt_path).write_text(subtitle_srt, encoding="utf-8")
        cmd.extend(["-vf", f"subtitles={srt_path}:force_style='FontSize=24,PrimaryColour=&HFFFFFF&'"])

    # 4. 混入 BGM（可选）
    if bgm_path:
        cmd.extend([
            "-i", bgm_path,
            "-filter_complex",
            "[0:a][1:a]amix=inputs=2:duration=first:dropout_transition=2[a]",
            "-map", "0:v", "-map", "[a]",
        ])

    cmd.append(output_path)

    subprocess.run(cmd, check=True, capture_output=True)
    return output_path
```

```python
# server.py — handle_project_render

def handle_project_render(self, project_id):
    project = load_project(project_id)
    if not project:
        return self._json_error(404, "项目不存在")

    # 收集所有分镜视频的本地路径
    shot_videos = []
    for episode in project.get("episodes", []):
        for shot in episode.get("shots", []):
            if shot.get("video_local_path"):
                shot_videos.append(shot["video_local_path"])

    if not shot_videos:
        return self._json_error(400, "没有可用的分镜视频")

    # 生成字幕
    from video_composer import generate_srt, compose_video
    subtitle_srt = generate_srt(project["episodes"], flatten_shots(project))

    # 输出路径
    output_dir = ensure_project_dir(project_id) / "output"
    output_path = str(output_dir / "final_output.mp4")

    # 合成（异步）
    import threading
    def do_compose():
        try:
            compose_video(shot_videos, output_path, subtitle_srt)
            project["post_production"]["final_output"] = f"/api/projects/{project_id}/output/final_output.mp4"
            project["status"] = "done"
            persist_project(project_id, project)
        except Exception as e:
            project["status"] = "failed"
            persist_project(project_id, project)

    threading.Thread(target=do_compose, daemon=True).start()

    project["status"] = "compositing"
    persist_project(project_id, project)

    return self._json_response({
        "message": "合成已开始",
        "status": "compositing",
    })
```

### 7.1 Phase A/B 错误处理

| 错误场景 | 处理方式 |
|---------|---------|
| 分镜图生成失败 | 自动重试 1 次；仍失败标记为 `error`，用户可手动重生 |
| 视频生成超时（>5min） | 标记为 `failed`，用户可重试 |
| 视频生成返回低质量 | 用户审核不通过 → 修改提示词 → 重新生成 |


### 7.5 分阶段合成流水线（断点续做）

合成过程分为 7 个阶段，系统跟踪每个阶段的产物，支持从失败阶段重做：

```
合成阶段流水线:

[1.拼接] → [2.转场] → [3.字幕] → [4.配音] → [5.BGM] → [6.混缩] → [7.渲染]
  ✅          ✅         ❌         ⏭          ⏭         ⏭          ⏭

→ 用户选择「仅重试失败阶段」→ 从 stage_3 的输入(stage_2 产物)开始重做
```

```python
# video_composer.py — 分阶段合成

import shutil
from pathlib import Path

COMPOSE_STAGES = ["concat", "transition", "subtitle", "voiceover", "bgm", "audio_mix", "render"]

def compose_video_staged(
    project_id: str,
    shot_videos: list[str],
    config: dict,
    resume_from: str | None = None,
) -> dict:
    """分阶段合成，支持断点续做"""
    output_dir = Path(ensure_project_dir(project_id)) / "output"
    stages_dir = output_dir / "stages"
    stages_dir.mkdir(parents=True, exist_ok=True)

    stages = COMPOSE_STAGES
    if resume_from and resume_from in stages:
        stages = stages[stages.index(resume_from):]

    current_input = None
    results = {"stages_completed": [], "stages_failed": None}

    for stage in stages:
        try:
            output = str(stages_dir / f"stage_{stage}_output.mp4")

            if stage == "concat":
                ffmpeg_concat(shot_videos, output)
            elif stage == "transition":
                ffmpeg_add_transitions(current_input, output, config.get("transition_style", "dissolve"))
            elif stage == "subtitle" and config.get("include_subtitles"):
                if config.get("subtitle_mode") == "burn":
                    ffmpeg_burn_subtitles(current_input, output, config["srt_path"])
                else:
                    shutil.copy(current_input, output)  # 外挂字幕不烧录
            elif stage == "voiceover" and config.get("include_voiceover"):
                ffmpeg_add_voiceover(current_input, output, config["voiceover_path"])
            elif stage == "bgm" and config.get("include_bgm"):
                ffmpeg_add_bgm(current_input, output, config["bgm_path"])
            elif stage == "audio_mix":
                ffmpeg_audio_mix(current_input, output)
            elif stage == "render":
                ffmpeg_final_render(current_input, output, config.get("output_quality", "1080p"))
            else:
                if current_input:
                    shutil.copy(current_input, output)

            current_input = output
            results["stages_completed"].append(stage)
        except Exception as e:
            results["stages_failed"] = {
                "stage": stage,
                "error": classify_compose_error(e),
            }
            break

    if not results["stages_failed"]:
        final_path = str(output_dir / "final_output.mp4")
        shutil.copy(current_input, final_path)
        results["final_output"] = final_path

    return results
```

```python
# server.py — handle_project_render（更新版）

def handle_project_render(self, project_id):
    project = load_project(project_id)
    body = json.loads(self.rfile.read(int(self.headers["Content-Length"])))

    # 合成配置
    compose_config = {
        "include_subtitles": body.get("include_subtitles", True),
        "subtitle_mode": body.get("subtitle_mode", "burn"),
        "include_voiceover": body.get("include_voiceover", True),
        "include_bgm": body.get("include_bgm", True),
        "bgm_source": body.get("bgm_source", "auto"),
        "output_quality": body.get("output_quality", "1080p"),
        "transition_style": body.get("transition_style", "dissolve"),
    }
    resume_from = body.get("resume_from_stage", None)

    shot_videos = collect_shot_videos(project)
    if not shot_videos:
        return self._json_error(400, "没有可用的分镜视频")

    import threading
    def do_compose():
        try:
            result = compose_video_staged(project_id, shot_videos, compose_config, resume_from)
            if result.get("final_output"):
                project["status"] = "done"
                project["post_production"]["final_output"] = result["final_output"]
            else:
                project["status"] = "compositing_failed"
                project["post_production"]["last_error"] = result["stages_failed"]
            project["post_production"]["compose_config"] = compose_config
            persist_project(project_id, project)
        except Exception as e:
            project["status"] = "compositing_failed"
            project["post_production"]["last_error"] = {"code": "UNKNOWN", "message": str(e)}
            persist_project(project_id, project)

    threading.Thread(target=do_compose, daemon=True).start()

    project["status"] = "compositing"
    persist_project(project_id, project)
    return self._json_response({"message": "合成已开始", "status": "compositing", "resume_from": resume_from})
```

### 8.2 版本存储

```
storage/projects/proj-xxx/versions/step5/
├── v001_20260328T140000.json    # 首次合成（调整字幕后重新合成前存档）
└── v002_20260328T143000.json    # 修改BGM后重新合成前存档
```

**注意**: 视频文件体积较大，版本快照**仅记录路径引用**，不复制视频文件。如果对应的视频文件被覆盖，旧版本标记为 `files_expired: true`。

最多保留 **5 个版本**。

### 8.3 回退说明

- 恢复版本时，只恢复合成配置和 shot 状态，不恢复视频文件本身
- 如果引用的视频文件已不存在，提示用户需要重新生成对应镜头
