### 3.2 提示词注入流程

```typescript
function buildScriptPrompt(project: WfProject): string {
  const style = project.style_config!;
  const shotCount = Math.ceil(style.duration_sec / style.shot_duration_sec);

  // 从风格提示词中提取关键词用于注入
  const styleKeywords = extractTopKeywords(style.compiled_style_prompt, 8);

  return SCRIPT_SYSTEM_PROMPT
    .replace("{COMPILED_STYLE_PROMPT}", style.compiled_style_prompt)
    .replace("{DURATION_SEC}", String(style.duration_sec))
    .replace("{ASPECT_RATIO}", style.aspect_ratio)
    .replace("{LANGUAGE}", style.language)
    .replace("{SHOT_COUNT}", String(shotCount))
    .replace("{SHOT_DURATION}", String(style.shot_duration_sec))
    .replaceAll("{STYLE_KEYWORDS}", styleKeywords);
}
```

### 4.2 JSON 提取与修复

```typescript
function extractJson(text: string): string {
  // 1. 尝试直接解析
  try { JSON.parse(text); return text; } catch {}

  // 2. 从 markdown 代码块中提取
  const codeBlock = text.match(/```(?:json)?\s*\n?([\s\S]*?)\n?```/);
  if (codeBlock) {
    try { JSON.parse(codeBlock[1]); return codeBlock[1]; } catch {}
  }

  // 3. 找 { ... } 范围
  const first = text.indexOf("{");
  const last = text.lastIndexOf("}");
  if (first !== -1 && last > first) {
    const candidate = text.slice(first, last + 1);
    try { JSON.parse(candidate); return candidate; } catch {}

    // 4. 尝试修复常见问题（尾部逗号等）
    const fixed = candidate
      .replace(/,\s*([}\]])/g, "$1")       // 尾部逗号
      .replace(/'/g, '"')                   // 单引号→双引号
      .replace(/(\w+)\s*:/g, '"$1":');      // 无引号 key
    try { JSON.parse(fixed); return fixed; } catch {}
  }

  throw new Error("无法从 AI 响应中提取有效 JSON");
}
```

### 4.3 校验函数

```typescript
function validateScriptAnalysis(data: ScriptAnalysis): void {
  const errors: string[] = [];

  if (!data.title) errors.push("缺少 title");
  if (!data.characters?.length) errors.push("缺少角色列表");
  if (!data.scenes?.length) errors.push("缺少场景列表");
  if (!data.episodes?.length) errors.push("缺少分集信息");

  for (const char of data.characters || []) {
    if (!char.char_id) errors.push(`角色 ${char.name} 缺少 char_id`);
    if (!char.three_view_prompts?.front) errors.push(`角色 ${char.name} 缺少正面提示词`);
  }

  for (const scene of data.scenes || []) {
    if (!scene.scene_id) errors.push(`场景 ${scene.name} 缺少 scene_id`);
    if (!scene.six_view_prompts?.front) errors.push(`场景 ${scene.name} 缺少正面提示词`);
  }

  if (errors.length > 0) {
    console.warn("剧本解析结果校验警告:", errors);
  }
}
```

## 七、错误处理

| 错误场景 | 处理方式 |
|---------|---------|
| LLM 返回非 JSON | 尝试 extractJson 修复；失败则提示用户「解析失败，请重试」 |
| LLM 超时（>120s） | 重试 1 次；仍然失败则切换到 `gpt-4o` 重试 |
| 返回缺少必要字段 | 标记缺失字段为红色，让用户手动补充 |
| 角色数过多（>10） | 警告：「角色过多可能影响一致性，建议控制在 5 个以内」 |
| 分镜数与预期差异大 | 警告：「预计 {n} 个分镜，实际 {m} 个」，可选择重新生成 |
| 流式连接中断 | 保留已接收内容，提供「继续生成」按钮 |
