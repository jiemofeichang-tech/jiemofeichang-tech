## 二、整体工作流

```
用户输入剧本 ──► 风格定调 ──► AI剧本解析 ──► 资产生成 ──► 视频合成
   Step 1         Step 2       Step 3        Step 4       Step 5
   (00-01)        (00-02)      (00-03)       (00-04)      (00-05)
```

### 状态机

```
draft → configured → scripting → script_parsed → designing →
assets_locked → storyboarding → filming → compositing → done
                                                  ↓
                                          compositing_failed
                                        （可调整参数后重试）
```

对应现有 `WfProject.status` 字段：
- `draft` — 新建项目，未填剧本
- `configured` — 风格配置完成（Step 2）
- `scripting` — 剧本解析中（Step 3）
- `script_parsed` — 剧本解析完成，待确认
- `designing` — 角色/场景资产生成中（Step 4）
- `assets_locked` — 资产锁定（Step 4 完成）
- `storyboarding` — 分镜图生成中（Step 5a）
- `filming` — 分镜视频生成中（Step 5b）
- `compositing` — 后期合成中（Step 5c）
- `compositing_failed` — 🆕 合成失败（可调整参数后重试 → compositing）
- `done` — 完成

## 五、版本历史机制

### 5.1 设计目标

用户在任何步骤点击「重新生成」时，当前结果自动存档为版本快照，可随时回退到之前满意的版本。

### 5.2 存储结构

```
storage/projects/proj-xxx/
├── project.json
├── versions/                    # 🆕 版本历史目录
│   ├── step2/                   # 风格配置版本
│   │   ├── v001_20260328T100000.json
│   │   └── v002_20260328T103000.json
│   ├── step3/                   # 剧本解析版本
│   │   └── v001_20260328T110000.json
│   ├── step4/                   # 资产生成版本
│   │   ├── v001_20260328T120000.json
│   │   └── v001_assets/         # 图片副本
│   └── step5/                   # 分镜/合成版本
│       └── v001_20260328T140000.json
├── assets/
└── output/
```

### 5.3 数据结构

```typescript
/** 版本快照文件内容 */
interface VersionSnapshot {
  version_id: string;          // "v001"
  step: "step2" | "step3" | "step4" | "step5";
  created_at: string;          // ISO 时间戳
  trigger: string;             // "regenerate" | "manual_save" | "before_restore"
  label?: string;              // 用户自定义标签（可选）
  data: Record<string, any>;   // 该步骤的产物快照
}

/** 版本索引条目（轻量，存在 project.json 中） */
interface VersionEntry {
  version_id: string;
  created_at: string;
  trigger: string;
  label?: string;
  snapshot_path: string;       // "versions/step3/v001_xxx.json"
  preview?: Record<string, any>; // 摘要信息（角色数、分镜数等）
}

/** project.json 新增字段 */
interface WfProject {
  // ... 现有字段 ...
  version_index: {             // 🆕 版本索引
    step2: VersionEntry[];
    step3: VersionEntry[];
    step4: VersionEntry[];
    step5: VersionEntry[];
  };
}
```

### 5.4 版本创建时机

| 步骤 | 触发条件 | 快照内容 | 版本上限 |
|------|---------|---------|---------|
| Step 1 | — | 不记录版本（原始输入不可逆） | — |
| Step 2 | 重选风格 / 手动编辑提示词 | `style_config` | 10 |
| Step 3 | 重新生成 / AI 对话修改 | `script.analysis`（角色+场景+分镜） | 10 |
| Step 4 | 重新生成视图 / 上传替代图 | `characters` + `scenes` + 图片复制 | 5 |
| Step 5 | 重新生成分镜/视频 / 重新合成 | `shots` 状态或 `post_production` | 5 |

### 5.5 核心操作函数

```typescript
const MAX_VERSIONS: Record<string, number> = {
  step2: 10, step3: 10, step4: 5, step5: 5
};

/** 创建版本快照 */
async function createVersionSnapshot(
  projectId: string,
  step: string,
  trigger: string,
  data: Record<string, any>,
  label?: string,
): Promise<VersionEntry> {
  const project = await wfGetProject(projectId);
  const versions = project.version_index?.[step] || [];

  // 生成版本号和文件名
  const num = String(versions.length + 1).padStart(3, "0");
  const ts = new Date().toISOString().replace(/[-:]/g, "").split(".")[0];
  const snapshotPath = `versions/${step}/v${num}_${ts}.json`;

  // 保存快照文件
  const snapshot: VersionSnapshot = {
    version_id: `v${num}`,
    step: step as any,
    created_at: new Date().toISOString(),
    trigger,
    label,
    data,
  };
  await saveVersionFile(projectId, snapshotPath, snapshot);

  // Step 4 的图片需要复制到版本目录
  if (step === "step4") {
    await copyAssetFiles(projectId, `versions/step4/v${num}_assets/`);
  }

  // 更新版本索引
  const entry: VersionEntry = {
    version_id: `v${num}`,
    created_at: snapshot.created_at,
    trigger,
    label,
    snapshot_path: snapshotPath,
    preview: buildPreview(step, data),
  };
  versions.push(entry);

  // 超出上限时淘汰最旧版本
  while (versions.length > MAX_VERSIONS[step]) {
    const oldest = versions.shift()!;
    await deleteVersionFile(projectId, oldest.snapshot_path);
  }

  // 持久化索引
  await wfUpdateProject(projectId, {
    version_index: { ...project.version_index, [step]: versions },
  });

  return entry;
}

/** 恢复版本 */
async function restoreVersion(
  projectId: string,
  step: string,
  versionId: string,
): Promise<void> {
  const project = await wfGetProject(projectId);
  const entry = project.version_index[step]?.find(v => v.version_id === versionId);
  if (!entry) throw new Error("版本不存在");

  // 恢复前先自动保存当前状态
  const currentData = extractStepData(project, step);
  await createVersionSnapshot(projectId, step, "before_restore", currentData);

  // 加载快照并应用
  const snapshot = await loadVersionFile(projectId, entry.snapshot_path);
  await applyStepData(projectId, step, snapshot.data);
}

/** 构建版本预览摘要 */
function buildPreview(step: string, data: Record<string, any>): Record<string, any> {
  switch (step) {
    case "step2": return { style: data.art_substyle };
    case "step3": return {
      characters_count: data.characters?.length || 0,
      scenes_count: data.scenes?.length || 0,
      shots_count: countShots(data.episodes),
    };
    case "step4": return { assets_count: countAssets(data) };
    case "step5": return { shots_done: countDoneShots(data) };
    default: return {};
  }
}
```
